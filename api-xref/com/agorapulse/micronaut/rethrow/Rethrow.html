18  package com.agorapulse.micronaut.rethrow;
19  
20  import io.micronaut.aop.Around;
21  import io.micronaut.context.annotation.Type;
22  
23  import java.lang.annotation.Documented;
24  import java.lang.annotation.ElementType;
25  import java.lang.annotation.Retention;
26  import java.lang.annotation.Target;
27  import java.util.function.Function;
28  
29  import static java.lang.annotation.RetentionPolicy.RUNTIME;
30  
31  /**
32   * Rethrows an exception as different one.
33   *
34   * You should set the <code>value</code> or the <code>as</code> property otherwise
35   * {@link RuntimeException} used as wrapper function.
36   *
37   * If <code>value</code> is set then <code>as</code> is ignored.
38   */
39  @Documented
40  @Retention(RUNTIME)
41  @Target({ElementType.TYPE, ElementType.METHOD})
42  @Around
43  @Type(RethrowInterceptor.class)
44  public @interface Rethrow {
45  
46      /**
47       * This position moves the {@link Rethrow} annotation around {@link io.micronaut.retry.annotation.Retryable} interceptor.
48       *
49       * If {@link io.micronaut.retry.annotation.Retryable} and {@link Rethrow} is used on the same method then then
50       * rethrowin the exception will happen after the last attempt from the {@link io.micronaut.retry.annotation.Retryable}
51       * annotation.
52       */
53      int RETHROW_POSITION = -200;
54  
55      /**
56       * @return Function or closure which coverts the thrown exception to new one. New exception must extend {@link RuntimeException}
57       */
58      Class<? extends Function<? extends Throwable, ? extends RuntimeException>> value() default RethrowAsRuntimeException.class;
59  
60      /**
61       * @return Type of the {@link RuntimeException} wrapping the original cause. The exception class must have constructor which accepts {@link Throwable}
62       */
63      Class<? extends RuntimeException> as() default RuntimeException.class;
64  
65      /**
66       * @return only handle exceptions assignable to given type
67       */
68      Class<? extends Throwable>[] only() default Throwable.class;
69  
70      /**
71       * @return Additional message for the rethrown {@link RuntimeException}. The exception class must have constructor accepting two arguments ({@link String} and {@link Throwable}
72       */
73      String message() default "";
74  
75  }
